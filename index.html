// --- 修正されたデータ管理ロジック --- const CSV_FILE_PATH = 'data.csv'; const STORAGE_KEY = 'history_quiz_miss_list'; // 保存キー名 let masterData = []; let currentQueue = []; let currentIndex = 0; let currentMode = ''; let isFlipped = false; // 1. LocalStorageから「ミス問のリスト」を取得する function getSavedMissList() { const saved = localStorage.getItem(STORAGE_KEY); // 保存形式: ["年号_出来事", "年号_出来事", ...] return saved ? JSON.parse(saved) : []; } // 2. 「ミス問」に該当するもの（streak < 0）だけを抽出して保存する function saveMissList() { const missList = masterData .filter(d => d.streak < 0) .map(d => `${d.yr}_${d.ev}`); localStorage.setItem(STORAGE_KEY, JSON.stringify(missList)); } // 履歴のリセット function clearProgress() { if(confirm("ミス問リストをすべて削除しますか？")) { localStorage.removeItem(STORAGE_KEY); location.reload(); } } // アプリ起動（CSV読み込み） window.addEventListener('DOMContentLoaded', async () => { const statusMsg = document.getElementById('status-msg'); try { const response = await fetch(CSV_FILE_PATH); if (!response.ok) throw new Error('CSVファイルが見つかりません'); const text = await response.text(); // 保存されているミス問の識別子リストを取得 const savedMissIdentifiers = getSavedMissList(); masterData = text.trim().split('\n').map(line => { const parts = line.split(','); if (parts.length < 2) return null; const yr = parts[0].trim(); const ev = parts[1].trim(); // CSV内のデータが、保存されたミス問リストに含まれているか判定 const isMiss = savedMissIdentifiers.includes(`${yr}_${ev}`); return { yr, ev, streak: isMiss ? -1 : 0 // ミス問なら-1、そうでなければ0 }; }).filter(d => d !== null); statusMsg.innerText = `CSV読み込み完了: ${masterData.length} 件`; document.getElementById('menu-area').classList.remove('hidden'); updateStatsDisplay(); } catch (err) { statusMsg.innerText = `エラー: ${err.message}`; statusMsg.style.color = 'red'; } }); // 結果を記録する関数（共通） function recordResult(isCorrect) { const item = currentQueue[currentIndex]; const masterItem = masterData.find(d => d.yr === item.yr && d.ev === item.ev); if (isCorrect) { // 正解時 if (masterItem.streak < 0) { // ミス問リストに入っている場合 masterItem.streak = 1; // 1回目正解 } else if (masterItem.streak > 0) { // すでに連続正解中 masterItem.streak++; } // 累計3回連続正解したら、0に戻す（＝ミス問リストから完全削除） if (masterItem.streak >= 3) { masterItem.streak = 0; } } else { // 不正解時は即座にミス問リスト入り（-1） masterItem.streak = -1; } // 常に最新の「ミス問状態」を保存 saveMissList(); currentIndex++; if (currentIndex < currentQueue.length) { document.getElementById('remaining-count').innerText = currentQueue.length - currentIndex; showQuestion(); } else { alert("終了しました！"); exitApp(); } }
